1a. SUM


    AREA SUM, CODE, READONLY
    ENTRY
START
    MOV R1, #10
    MOV R2, #0
LOOP
    ADD R2, R2, R1
    SUBS R1, #0X01
    BNE LOOP
STOP
    B STOP
END

3. STEPPER MOTOR


#include 

void clock_wise(void);
void anti_clock_wise(void);

unsigned long int var1, var2;
unsigned int i = 0, j = 0, k = 0;

int main(void)
{
    LPC_PINCON->PINSEL4 = 0x00000000;
    LPC_GPIO2->FIODIR = 0x0000000F;

    while(1)
    {
        for(j = 0; j < 50; j++)
        {
            clock_wise();
        }

        for(k = 0; k < 65000; k++);

        for(j = 0; j < 50; j++)
        {
            anti_clock_wise();
        }

        for(k = 0; k < 65000; k++);
    }
}

void clock_wise(void)
{
    var1 = 0x00000001;

    for(i = 0; i <= 3; i++)
    {
        LPC_GPIO2->FIOCLR = 0x0000000F;
        LPC_GPIO2->FIOSET = var1;
        var1 = var1 << 1;

        for(k = 0; k < 150000; k++);
    }
}

void anti_clock_wise(void)
{
    var1 = 0x00000008;

    for(i = 0; i <= 3; i++)
    {
        LPC_GPIO2->FIOCLR = 0x0000000F;
        LPC_GPIO2->FIOSET = var1;
        var1 = var1 >> 1;

        for(k = 0; k < 150000; k++);
    }
}

4. DC MOTOR


#include 

void Clock_Wise(void);
void AClock_Wise(void);

unsigned long i;

int main(void)
{
    LPC_PINCON->PINSEL1 = 0x00000000;
    LPC_PINCON->PINSEL3 = 0x00000000;
    LPC_GPIO0->FIODIR |= 0x04000000;
    LPC_GPIO1->FIODIR |= 0x01000000;
    while(1)
    {
        Clock_Wise();
        for(i = 0; i < 300000; i++);
        AClock_Wise();
        for(i = 0; i < 300000; i++);
    }
}

void Clock_Wise(void)
{
    LPC_GPIO1->FIOCLR = 0x01000000;
    for(i = 0; i < 10000; i++);
    LPC_GPIO0->FIOSET = 0x04000000;
    LPC_GPIO1->FIOSET = 0x01000000;
}

void AClock_Wise(void)
{
    LPC_GPIO1->FIOCLR = 0x01000000;
    for(i = 0; i < 10000; i++);
    LPC_GPIO0->FIOCLR = 0x04000000;
    LPC_GPIO1->FIOSET = 0x01000000;
}

6. PWM


#include 

void delay(unsigned int count) {
    unsigned int i;

    for (i = 0; i < count; i++);
}

int main(void) {
    unsigned int duty = 0;
    unsigned char up_down = 1;
    SystemInit();
    SystemCoreClockUpdate();
    LPC_PINCON->PINSEL7 &= ~(0x3 << 18);
    LPC_PINCON->PINSEL7 |= (0x2 << 18);
    LPC_SC->PCONP |= (1 << 6);
    LPC_SC->PCLKSEL0 &= ~(0x3 << 12);
    LPC_PWM1->PR = 0;
    LPC_PWM1->MR0 = 3000;
    LPC_PWM1->MR2 = 0;
    LPC_PWM1->MCR = (1 << 1);
    LPC_PWM1->PCR = (1 << 10);
    LPC_PWM1->TCR = (1 << 0) | (1 << 3);
    while (1) {

        if (up_down == 1) {

            duty += 30;

            if (duty >= 2900) {

                up_down = 0;
            }

        } else {

            duty -= 30;

            if (duty <= 30) {

                up_down = 1;
            }
        }

        LPC_PWM1->MR2 = duty;

        LPC_PWM1->LER = (1 << 2);

        delay(4000);
    }
}

7. INTERRUPT


#include 

void EINT3_IRQHandler(void);

unsigned char int3_flag = 0;

int main(void)
{
    LPC_PINCON->PINSEL4 |= 0x04000000;
    LPC_GPIO2->FIODIR = 0x00001000;
    LPC_GPIO2->FIOSET = 0x00001000;
    LPC_SC->EXTINT = 0x00000008;
    LPC_SC->EXTMODE = 0x00000008;
    LPC_SC->EXTPOLAR = 0x00000000;
    NVIC_EnableIRQ(EINT3_IRQn);
    while(1);
}

void EINT3_IRQHandler(void)
{
    LPC_SC->EXTINT = 0x00000008;

    if (int3_flag == 0x00)
    {
        LPC_GPIO2->FIOCLR = 0x00001000;

        int3_flag = 0xFF;
    }
    else
    {
        LPC_GPIO2->FIOSET = 0x00001000;

        int3_flag = 0x00;
    }
}
